global _start

section .data
	
	align 2
	hello: db 'Hello world!', 0xa
	helloLen: equ $-hello

section .bss

section .text

	_start:

	mov ebx, 0x1
	mov ecx, hello
	mov edx, helloLen
	int 0x80
